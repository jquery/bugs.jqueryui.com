Example file to show our usage of tabs

<!DOCTYPE html PUBLIC "-//W3C//DTD HTML 4.01//EN" "http://www.w3.org/TR/html4/strict.dtd">
<html>
<head>
<script type="text/javascript" language="JavaScript" src="js/jquery/jquery.js"></script>     
<script src="js/jquery/jquery-ui-personalized-1.5.1.js" type="text/javascript"></script>   
<script language="javascript" type="text/javascript">  
		$(document).ready(function(){
			var tabs=$("#tabs").tabs({remote: true, selected: null});
			tabs.tabs('select',1);
		});
</script>
</head>
<body>
<form>

<div id="scrollHeader">
<div id="tabs">
	<ul> 
		<li><a title="tabContent"
			href="ajax action">			
			<span>tab 1</span>
			</a></li>
		<li><a title="tabContent"
			href="ajax action">
			<span>tab 2</span></a></li>
	</ul>
	</div>
</div>
<div id="scrollBody">
<div id="tabContent">
<!-- Tab content goes here -->
</div>
</div>
<div id="scrollFooter">
<ul class="commandButtons">
		<li>
		<button onClick="submitForm();" >Save</button>
		</li>
		<li>
		<button onClick="javascript:document.forms[0].reset();" >Reset</button>
		</li>
</ul>
</div>
</form>
</BODY>
</HTML>
