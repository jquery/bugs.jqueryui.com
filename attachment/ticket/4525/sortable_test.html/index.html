Delete of Currently Selected Tab Deletes Two Tabs

<!doctype html>
<html lang="en">
<head>
	<title>jQuery UI Tabs - Sortable</title>
	<link type="text/css" href="../../themes/base/ui.all.css" rel="stylesheet" />
	<script type="text/javascript" src="../../jquery-1.3.2.js"></script>
	<script type="text/javascript" src="../../ui/ui.core.js"></script>
	<script type="text/javascript" src="../../ui/ui.sortable.js"></script>
	<script type="text/javascript" src="../../ui/ui.tabs.js"></script>
	<link type="text/css" href="../demos.css" rel="stylesheet" />
	<script type="text/javascript">
	$(function() {
	
	$("#tabs").tabs().find(".ui-tabs-nav").sortable({axis:'x'});
		
	jQuery('.ui-tabs-nav li img').live('click', function(event)
	{
		var ourMrX	= jQuery(this);
		var ourMrId	= ourMrX.closest('li').find('a').attr('href');
		
		alert('click');
		
		jQuery(ourMrId).hide('slow', function()
		{
			alert('hide');
			jQuery("#tabs").tabs('remove', (jQuery('#tabs ul:first li').index(jQuery(ourMrX).closest('li')) - 1));
			
			event.stopPropagation();
			
			return;
		});
		
		event.stopPropagation();
	});
	
	jQuery('#add_tab').click(function()
	{
		var tabName = prompt("Label for New Tab:", '');
		
		if ( ! tabName || tabName == null)
		{
			return false; 
		}
		
		tabId = 'tabs-' + (jQuery('#tabs ul:first li').length);
		
		// Add Tab		
		jQuery("#tabs").tabs('add', '#' + tabId, tabName);
		
		// Add delete image
		jQuery('a[href=#' + tabId + ']').after('<img src="nomatter.png" width="15" height="15" />');
		
		// Select
		jQuery("#tabs").tabs('select', '#' + tabId);
		
		return false;
	});
		
		
	});
	</script>
</head>
